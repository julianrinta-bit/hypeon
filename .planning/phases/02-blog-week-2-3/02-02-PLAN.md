```tsx
interface Post {
  title: string
  description: string
  date: string
  updated?: string
  permalink: string
  cover: { src: string; width: number; height: number }
}

export default function BlogJsonLd({ post }: { post: Post }) {
  const jsonLd = {
    '@context': 'https://schema.org',
    '@type': 'BlogPosting',
    headline: post.title,
    description: post.description,
    datePublished: post.date,
    dateModified: post.updated ?? post.date,
    image: {
      '@type': 'ImageObject',
      url: `https://hypeon.media${post.cover.src}`,
      width: post.cover.width,
      height: post.cover.height,
    },
    author: {
      '@type': 'Organization',
      name: 'Hype On Media',
      url: 'https://hypeon.media',
    },
    publisher: {
      '@type': 'Organization',
      name: 'Hype On Media',
      url: 'https://hypeon.media',
      logo: {
        '@type': 'ImageObject',
        url: 'https://hypeon.media/logo.svg',
      },
    },
    mainEntityOfPage: {
      '@type': 'WebPage',
      '@id': `https://hypeon.media${post.permalink}`,
    },
  }

  return (
    <script
      type="application/ld+json"
      dangerouslySetInnerHTML={{ __html: JSON.stringify(jsonLd) }}
    />
  )
}
```

Key decisions:
- `author` is Organization, not Person (faceless brand).
- `publisher.logo` references `/logo.svg` — verify this file exists in `public/`. If it does not, use the OG image URL instead.
- `dateModified` falls back to `datePublished` when `updated` is not set.
- Image URL is absolute (`https://hypeon.media` + relative path from velite).
</action>

```ts
import type { MetadataRoute } from 'next'
import { posts } from '#velite'

export default function sitemap(): MetadataRoute.Sitemap {
  const blogEntries: MetadataRoute.Sitemap = posts.map(post => ({
    url: `https://hypeon.media${post.permalink}`,
    lastModified: new Date(post.updated ?? post.date),
    changeFrequency: 'monthly',
    priority: 0.7,
  }))

  return [
    {
      url: 'https://hypeon.media',
      lastModified: new Date(),
      changeFrequency: 'weekly',
      priority: 1,
    },
    {
      url: 'https://hypeon.media/blog',
      lastModified: new Date(),
      changeFrequency: 'weekly',
      priority: 0.8,
    },
    ...blogEntries,
  ]
}
```

Changes from current:
- Imports `posts` from `#velite`
- Adds `/blog` index page with priority 0.8
- Dynamically adds all published blog post URLs with priority 0.7
- Each post URL uses `post.updated ?? post.date` as lastModified
- `changeFrequency: 'monthly'` for individual posts (content is evergreen)
</action>
