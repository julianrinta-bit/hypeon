# Plan 2.1 — MDX Pipeline + Pages (Wave 1)

## Objective

Stand up the full velite-powered MDX blog pipeline: install dependencies, configure velite with frontmatter schema, integrate with Next.js 16 + Turbopack, create blog index and article pages with SSG, and add prose styling matching the dark theme + gold accent. A single placeholder article verifies the pipeline end-to-end.

## Constraints

- Next.js 16.2 with Turbopack: must use the top-level `import('velite')` pattern in `next.config.ts`, NOT the webpack plugin.
- `params` is a Promise in Next.js 16 dynamic routes: both the page component and `generateMetadata` must `await params`.
- `MDXContent` must be `'use client'` because it uses `new Function()`.
- Faceless brand: author is always "Hype On Media" (Organization), never a person.
- All CSS goes in `globals.css` (no CSS Modules, no Tailwind).
- Design tokens from `:root`: `--bg: #09090b`, `--accent: #FFD300`, `--white: #f0f0ec`, `--font-display`, `--font-body`, `--font-mono`.

```ts
import rehypePrettyCode from 'rehype-pretty-code'
import rehypeSlug from 'rehype-slug'
import rehypeAutolinkHeadings from 'rehype-autolink-headings'
import { defineCollection, defineConfig, s } from 'velite'

const posts = defineCollection({
  name: 'Post',
  pattern: 'blog/**/*.mdx',
  schema: s
    .object({
      title: s.string().max(120),
      slug: s.slug('blog'),
      description: s.string().max(300),
      date: s.isodate(),
      updated: s.isodate().optional(),
      cover: s.image(),
      coverAlt: s.string(),
      tags: s.array(s.string()).default([]),
      draft: s.boolean().default(false),
      metadata: s.metadata(),
      excerpt: s.excerpt(),
      toc: s.toc(),
      code: s.mdx(),
    })
    .transform(data => ({
      ...data,
      permalink: `/blog/${data.slug}`,
    })),
})

export default defineConfig({
  root: 'content',
  output: {
    data: '.velite',
    assets: 'public/static',
    base: '/static/',
    name: '[name]-[hash:6].[ext]',
    clean: true,
  },
  collections: { posts },
  markdown: {
    rehypePlugins: [
      rehypeSlug,
      [rehypeAutolinkHeadings, { behavior: 'wrap' }],
      rehypePrettyCode,
    ],
  },
  prepare: ({ posts }) => {
    const published = posts.filter(
      p => process.env.NODE_ENV !== 'production' || !p.draft
    )
    posts.length = 0
    posts.push(...published)
  },
})
```

Key schema fields:
- `s.slug('blog')` generates slug from directory name
- `s.image()` processes cover images to `public/static/` with content hashes
- `s.metadata()` provides `wordCount` and `readingTime`
- `s.mdx()` compiles MDX to a function body string
- `.transform()` adds `permalink` field
- `prepare` filters drafts in production builds
</action>

```tsx
import * as runtime from 'react/jsx-runtime'
import type { ComponentType } from 'react'

const useMDXComponent = (code: string) => {
  const fn = new Function(code)
  return fn({ ...runtime }).default
}

interface MDXProps {
  code: string
  components?: Record<string, ComponentType>
}

export const MDXContent = ({ code, components }: MDXProps) => {
  const Component = useMDXComponent(code)
  return <Component components={components} />
}
```

```tsx
import { posts } from '#velite'
import { notFound } from 'next/navigation'
import Image from 'next/image'
import type { Metadata } from 'next'
import { MDXContent } from '@/components/blog/MDXContent'
import InlineCTA from '@/components/blog/InlineCTA'

// SSG: pre-render all slugs at build time
export function generateStaticParams() {
  return posts.map(post => ({ slug: post.slug }))
}

// Dynamic metadata per article
// Next.js 16: params is a Promise — must await
export async function generateMetadata({
  params,
}: {
  params: Promise<{ slug: string }>
}): Promise<Metadata> {
  const { slug } = await params
  const post = posts.find(p => p.slug === slug)
  if (!post) return {}

  return {
    title: `${post.title} — Hype On Media`,
    description: post.description,
    openGraph: {
      title: post.title,
      description: post.description,
      type: 'article',
      publishedTime: post.date,
      modifiedTime: post.updated ?? post.date,
      url: `https://hypeon.media${post.permalink}`,
      images: [{ url: post.cover.src }],
    },
    twitter: { card: 'summary_large_image' },
    alternates: { canonical: post.permalink },
  }
}

// Next.js 16: params is a Promise — must await
export default async function BlogPostPage({
  params,
}: {
  params: Promise<{ slug: string }>
}) {
  const { slug } = await params
  const post = posts.find(p => p.slug === slug)
  if (!post) notFound()

  return (
    <main id="main-content">
      <article className="blog-article">
        <header className="blog-article__header">
          <div className="blog-article__meta">
            <time dateTime={post.date}>
              {new Date(post.date).toLocaleDateString('en-US', {
                year: 'numeric', month: 'long', day: 'numeric',
              })}
            </time>
            <span>{post.metadata.readingTime} min read</span>
          </div>
          <h1>{post.title}</h1>
          <p className="blog-article__description">{post.description}</p>
        </header>

        <Image
          src={post.cover.src}
          alt={post.coverAlt}
          width={post.cover.width}
          height={post.cover.height}
          className="blog-article__cover"
          priority
        />

        <div className="blog-article__body prose">
          <MDXContent code={post.code} components={{ InlineCTA }} />
        </div>
      </article>
    </main>
  )
}
```

Critical Next.js 16 detail: `params` is typed as `Promise<{ slug: string }>` and must be awaited in both the page component and `generateMetadata`. The component is `async` for this reason.

`generateStaticParams` pre-renders all article slugs at build time (SSG). Returns plain objects, not Promises.
</action>
